001package com.cosm.client.model;
002
003import com.cosm.client.requester.CollectionUtil;
004
005/**
006 * Marker interface for all resources that can be requested from Cosm API
007 * directly
008 * 
009 * @author s0pau
010 * 
011 */
012public interface ConnectedObject<T>
013{
014        String getIdString();
015
016        /**
017         * @param other
018         * @return true, if all member fields are equal by equal() method
019         *         (collections are considered equal if the members of the
020         *         collection's equal() returns true using
021         *         {@link CollectionUtil#equal(java.util.Collection, java.util.Collection)}
022         *         ; false otherwise
023         */
024        boolean memberEquals(T other);
025}
