001package com.cosm.client.requester;
002
003import java.util.List;
004
005public class StringUtil
006{
007        public static boolean isNullOrEmpty(String toCheck)
008        {
009                if (toCheck == null || toCheck.length() == 0)
010                {
011                        return true;
012                }
013                return false;
014        }
015
016        /**
017         * 
018         * @param list
019         * @return comma delimited string of elements in list, {@see toString} which
020         *         is used to convert each element
021         */
022        public static String toCommaDelimitedString(List list)
023        {
024                StringBuilder retval = null;
025                boolean isFirstToken = true;
026                for (Object o : list)
027                {
028                        if (o == null)
029                        {
030                                continue;
031                        }
032
033                        String token = null;
034                        token = toString(o);
035
036                        if (token.length() > 0)
037                        {
038                                if (isFirstToken)
039                                {
040                                        isFirstToken = false;
041                                        retval = new StringBuilder();
042
043                                } else
044                                {
045                                        retval.append(",");
046                                }
047                                retval.append(token);
048                        }
049                }
050
051                return retval == null ? null : retval.toString();
052        }
053
054        /**
055         * 
056         * @param o
057         * @return string of the object
058         * @exception UnsupportedOperationException
059         *                if the object is not String/Number/Boolean of a list of
060         *                String/Number/Boolean
061         */
062        public static String toString(Object o)
063        {
064                if (o instanceof String)
065                {
066                        return (String) o;
067                } else if (o instanceof Boolean)
068                {
069                        return String.valueOf(o);
070                } else if (Number.class.isAssignableFrom(o.getClass()))
071                {
072                        return String.valueOf((Number) o);
073                } else if (List.class.isAssignableFrom(o.getClass()))
074                {
075                        return toCommaDelimitedString((List) o);
076                } else
077                {
078                        throw new ParseToObjectException(String.format("Cannot create comma delimited string for %s", o.getClass()));
079                }
080        }
081}
